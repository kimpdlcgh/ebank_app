rules_version = '2';

// Firebase Storage Security Rules for Banking Dashboard
service firebase.storage {
  match /b/{bucket}/o {
    
    // Allow authenticated users to upload branding assets
    match /branding/{allPaths=**} {
      allow read: if true; // Public read for branding assets
      allow write: if request.auth != null 
        && request.auth.token.role in ['SUPER_ADMIN', 'ADMIN']
        && resource.size < 5 * 1024 * 1024 // 5MB limit
        && request.resource.contentType.matches('image/.*');
    }
    
    // Allow authenticated users to manage their profile images
    match /profiles/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null 
        && (request.auth.uid == userId || request.auth.token.role in ['SUPER_ADMIN', 'ADMIN'])
        && resource.size < 2 * 1024 * 1024; // 2MB limit for profile images
    }
    
    // Allow admins to manage document uploads
    match /documents/{allPaths=**} {
      allow read, write: if request.auth != null 
        && request.auth.token.role in ['SUPER_ADMIN', 'ADMIN']
        && resource.size < 10 * 1024 * 1024; // 10MB limit for documents
    }
    
    // Deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
